// sendToPatient 함수 내부를 아래처럼 순서대로 배치했습니다. // "팔"이나 "손"이 들어간 문장은 무조건 운동 지시(commK)로 가장 먼저 걸리게 순서를 바꿨습니다. function sendToPatient() { const text = document.getElementById("studentMsg").value.trim(); if(!text) return; appendMsg(text, "student"); document.getElementById("studentMsg").value = ""; const data = aiResponses[activeCaseIdx]; if(!data) return; // 키워드 정의 const commK = ["손", "팔", "다리", "발", "움직", "펴", "구부", "악수", "상지", "하지", "지시", "명령"]; const painK = ["통증", "자극", "꼬집", "누르", "압박", "흉골", "손톱", "아프", "아픈"]; const whereK = ["어디", "병원", "날짜", "오늘", "며칠", "이름", "성함", "연도", "장소", "시간", "누구", "기억", "여기가"]; const callK = ["눈", "떠", "들리", "깨워", "환자분", "여보세요", "일어나"]; let reply = data.default; // 핵심 수정: commK(운동지시)를 가장 먼저 확인하여 다른 키워드에 뺏기지 않게 함 if (commK.some(k => text.includes(k))) { let userReq = text.replace(/주세요|해봐요|보세요|어디/g, "").trim(); reply = data.commandBase.replace("{REQ}", userReq); } else if (painK.some(k => text.includes(k))) { reply = data.pain; } else if (whereK.some(k => text.includes(k))) { reply = data.where; } else if (callK.some(k => text.includes(k))) { reply = data.call; } setTimeout(() => appendMsg(reply, "patient"), 400); }
